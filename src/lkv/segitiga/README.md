## ðŸ”° Pseudocode

```
BEGIN
    DECLARE ALAS, TINGGI, SISI_A, SISI_B, SISI_C, LUAS, KELILING AS FLOAT

    INPUT ALAS
    INPUT TINGGI
    
    INPUT SISI_A
    INPUT SISI_B
    INPUT SISI_C

    SET LUAS TO (ALAS * TINGGI) / 2
    SET KELILING TO SISI_A + SISI_B + SISI_C

    OUTPUT LUAS
    OUTPUT KELILING
END
```
